## 受控状态

<Playground>
  {() => {

    class Demo extends React.PureComponent {
        constructor(props){
            super(props)
            this.state = {
                dataSource: {}
            }
            this.onValueChange = this.onValueChange.bind(this)
        }

        onValueChange(params) {
            this.setState({
                dataSource: params.values
            })
        }

        render () {
            return <>
               <Form
                controlNumberOneRow={3}
                controls={[
                    {
                    control: {
                        label: 'name',
                    },
                    field: 'name',
                    input: {
                        type: 'input',
                    },
                    decorator: {
                        rules: [
                        {
                            required: true,
                            message: 'name must be exsit',
                        },
                        ],
                    },
                    },
                    {
                    control: {
                        label: 'age',
                    },
                    field: 'age',
                    input: {
                        type: 'number',
                        precision: 2,
                        format: '0,0.00',
                    },
                    },
                    {
                    control: {
                        label: 'dell',
                    },
                    field: 'dell',
                    input: {
                        type: 'select',
                        decorator: [
                        {
                            label: 'name1',
                            value: 'name1',
                        },
                        ],
                    },
                    },
                ]}
                dataSource={this.state.dataSource}
                onValueChange={this.onValueChange}
                />
            </>
        }
    }

    return <Demo />

}}

</Playground>
