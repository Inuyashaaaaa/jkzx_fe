## Feature

## Install

## 部分功能展示

<Playground>
  {() => {
    const item = (title, print, result) => {
      return (
        <div>
          <p>{title}</p>
          <p style={{marginLeft:20,fontSize:15}}>{print} = {result}</p>
        </div>
      )
    }
    return (
      <div>
        { item('进行表达式运算:', '2+6*9-3', mathTy.express('2+6*9-3')) }
        { item('进行开方:', '16开方', mathTy({method: 'sqrt', params: [16]})) }
        { item('保留小数点后3位:', 'e', mathTy({method: 'round', params: [Math.E, 3 ]})) }
      </div>
    );
      
  }}
</Playground>
